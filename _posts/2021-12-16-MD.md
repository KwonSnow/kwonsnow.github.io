# Thermostat
VX[i] = VX[i] * scale  
VY[i] = VY[i] * scale  
VZ[i] = VZ[i] * scale  
## Velocity rescaling
scale = sqrt(T_REF/T_BULK)  
## Berendsen
scale = sqrt(1+DT*((T_REF/T_BULK)-1)/TAU)
